Database Backup Specific

#!/bin/bash

# Check disk space
AVAILABLE_SPACE=$(df -h /backup | tail -1 | awk '{print $4}' | sed 's/G//')
if (( $(echo "$AVAILABLE_SPACE < 5" | bc -l) )); then
    echo "Insufficient disk space" && curl -fsS -m 10 --retry 5 {URL}/fail
    exit 1
fi

# Execute database backup
mysqldump -u username -p password database_name > /backup/db_$(date +%Y%m%d).sql && curl -fsS -m 10 --retry 5 {URL} || curl -fsS -m 10 --retry 5 {URL}/fail
